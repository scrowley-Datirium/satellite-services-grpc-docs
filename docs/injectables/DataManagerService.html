import { Injectable } from '@nestjs/common';
import path = require('path');
import { Subject } from 'rxjs';
import { FileSystemStorage, StreamResponse } from 'send-stream';
import { Certs } from '../interfaces';

import { DataManagerData, Pm2Packet, Pm2Response } from '../interfaces/pm2.interface';
import Log from '../lib/logger';

// Service that handles all the communication with the main service
@Injectable()
export class DataManagerService {
    // Location of scidap projects directory
    public scidapHome: string;
    // Token that represents this satellite
    public satelliteToken: string;
    // Public key of API that is used to verify JWTs that hit endpoints
    public publicKey: string;
    
    private stopServers: () => Promise<any>;
    private startupListener$: Subject<Boolean> = new Subject();
    public serverKey: string;
    public serverCert: string;
    public serverPort: string; // http server, https server is +1

    // FileSystem root, updated whenever we update scidapHome
    public fsRoot = { projects: '', labs: '', local: '' };
    public storage: FileSystemStorage;
    public localFiles: any;
    private _fsKeys = ['localFiles', 'scidapHome'];

    private _keysToUpdate = ['scidapHome', 'satelliteToken', 'publicKey', 'serverKey', 'serverCert', 'serverPort', 'localFiles'];

    constructor() {
        process.on('message', (packet: Pm2Packet) => {
            Log.info('Message from main service received');
            Log.info(JSON.stringify(packet));
            this._processMessage(packet);
        });
    }

    private _processMessage(packet: Pm2Packet) {
        const keysChanged = [];
        const response: Pm2Response = { updated: keysChanged };
        try {
            const parsed: DataManagerData = JSON.parse(packet.data);
            Log.info(`${parsed}`);
            for (let key of this._keysToUpdate) {
                Log.info(key, parsed[key], !!parsed[key]);
                if (!!parsed[key]) {
                    this[key] = parsed[key];
                    keysChanged.push(key);
                }
                Log.info(this[key]);
            }

            if (keysChanged.length === 0) {
                throw new Error('No keys were updated');
            }
            // If we updated a key that is prefixed with server, restart the server
            if (keysChanged.some((k) => k.includes('server'))) this.startupListener$.next(true);
            // If we updated a key that has to deal with the filesystem update that
            if (keysChanged.some((k) => this._fsKeys.includes(k))) this._updateHome();
            Log.info(`Succesfully updated: ${keysChanged.join(', ')}`);
        } catch (err) {
            const msg = `Error parsing packet from main service: ${err}`;
            Log.error(msg);
            response.err = true;
            response.msg = msg;
        } finally {
            this.sendMessage(response);
        }
    }

    public sendMessage(message: Pm2Response) {
        process.send({
            type : 'process:msg',
            data : {
              ...message
            },
        })          
    }

    private _updateHome() {
        // this._localFiles = this.configService.get('remotes').localfiles;
        this.fsRoot.projects = path.resolve(this.scidapHome, 'projects');
        this.fsRoot.labs = path.resolve(this.scidapHome, 'laboratories');
        const localFiles = this.localFiles || { path: '' };
        this.fsRoot.local = path.resolve('/', localFiles.path);
        this.storage = new FileSystemStorage(this.scidapHome);
        // };
    // } catch (err) {
    //     Log.debug(err);
    //     Log.info('No Local Files Defined');
    //     this._fsRoot = {
    //         projects: path.resolve(this.configService.get('systemRoot'), 'projects'),
    //         labs: path.resolve(this.configService.get('systemRoot'), 'laboratories'),
    //         local: '',
    //     };

    }

    /**
     * Provide a hook for main.ts to start and stop servers programatically
     * @param startUpFunction Function that starts the Servers
     * @param stopServers Function that stops the Servers
     */
    public subscribeToStartup(startUpFunction: (certs: Certs, port: string) => void, stopServers: () => Promise<string[]>): void {
        this.startupListener$.subscribe(() => {
            const certs = {
                key: this.serverKey,
                cert: this.serverCert,
            };
            startUpFunction(certs, this.serverPort)
        },(err) => this.sendMessage({ err: true, msg: err }),
        );
        // Wrap function in try catch to keep from breaking
        this.stopServers = async () => {
            const response: Pm2Response = {};
            try {
                const msg = await stopServers();
                response.msg = msg.join(', ');
            } catch (err) {
                const msg = `Error Shutting down Servers: ${err}`;
                response.msg = msg;
                response.err = err;
            } finally {
                this.sendMessage(response);
            }
        };
    }
    
}
