import { Injectable } from '@nestjs/common';
import { Observable, Subject } from 'rxjs';
import { reduce, switchMap } from 'rxjs/operators';
import * as csv from 'fast-csv';
import fs = require('fs');
import path = require('path');

import Log from '../lib/logger';
import { VisualPluginRequest, VisualPluginResponse } from 'src/interfaces';
import { ConfigService } from '@nestjs/config';

@Injectable()
export class ReadFileService {
    private _events$ = new Subject<VisualPluginResponse>();
    constructor (private configService: ConfigService) {}
    public get events$() {
        return this._events$;
    }

    /**
     * Called at the start of satellite to feed it the stream of requests
     * @param $stream stream that represents requests for information about visual plugins
     */
    public startListening(stream$: Subject<VisualPluginRequest>) {
        stream$
            .pipe(switchMap((visualPluginRequest: VisualPluginRequest) => this._requestFileData(visualPluginRequest)))
            .subscribe((visualPluginResponse: VisualPluginResponse) => {
                visualPluginResponse && visualPluginResponse.error ? Log.error(visualPluginResponse) : Log.debug(visualPluginResponse);
                this._events$.next(visualPluginResponse);
            });
    }

    /**
     * Read a given file as a stream
     * @param filePath location of the file
     * @param options options to read the file
     */
    private _csvDataStream(filePath: string, options?): Observable<any> {
        let targetPath;
        if (filePath.includes(this.configService.get('systemRoot'))) {
            targetPath = filePath;
        } else {
            targetPath = path.join(this.configService.get('systemRoot'), filePath);
        }
        const file$ = fs.createReadStream(targetPath, { encoding: 'utf8' }).pipe(csv.parse(options));
        return new Observable((observer) => {
            file$.on('data', (chunk) => observer.next(chunk));
            file$.on('end', () => observer.complete());
            file$.on('close', () => observer.complete());
            file$.on('error', (err) => observer.error(err));
        });
    }

    /**
     * Parse the information before reading it
     *  TODO: Should this be an absolute path yet?
     * @param filePath location of the file
     * @param fields fields to create the pie chart
     */
    private _requestFileData(visualPluginRequest: VisualPluginRequest): Observable<any> | any {
        const visualPluginResponse: VisualPluginResponse = {
            sampleId: visualPluginRequest.sampleId,
            visualPlugin: {
                [visualPluginRequest.type]: {
                    ...visualPluginRequest.visualPlugins,
                    data: visualPluginRequest.data.map(() => []),
                },
            },
        };

        const reqFields = visualPluginRequest.data.map((item) => {
            const num = item.startsWith('$') ? item.slice(1) : item;
            return parseInt(num) - 1;
        });
        const clean = (x) => (isNaN(x) ? x : parseInt(x));
        const options = {
            delimiter: '\t',
            headers: false,
        };
        return this._csvDataStream(visualPluginRequest.filePath, options).pipe(
            reduce((response, curr) => {
                reqFields.forEach((item, i) => response.visualPlugin[visualPluginRequest.type].data[i].push(clean(curr[item])));
                return response;
            }, visualPluginResponse),
        );
    }
}
