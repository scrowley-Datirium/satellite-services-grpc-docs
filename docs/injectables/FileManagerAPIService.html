    /**
     * Get all the files in a directory
     * @param filePath path to file
     * @param rootPath rootPath with projects or laboratories
     */
    private async _getFiles(filePath: string): Promise<string[]> {
        Log.info('check get files', filePath);
        try {
            const upload_path = filePath;
            await fsPromises.mkdir(upload_path, { recursive: true });
        } catch (err) {
            Log.debug(`Accessing ${filePath}`);
        }

        try {
            const files: string[] = await fsPromises.readdir(filePath);
            return files;
        } catch (err) {
            Log.error(`Failed to get files: ${err}`);
            throw err;
        }
    }

    /**
     * Get information about the CWD
     * @param data information about the cwd from the file manager
     * @param filePath path to the cwd
     * @param rootPath rootPath with projects or laboratories
     */
    private async _fileManagerDirectoryContent(data: FileInformation[], filePath: string, rootPath: string): Promise<FileInformation> {
        try {
            Log.info('File Manager Directory Contents', data);
            const stats = await fsPromises.stat(filePath);
            let filterPath: string;
            if (data[0]?.filterPath) {
                filterPath = data[0].filterPath;
            } else {
                filterPath = data.length > 0 ? this.utilitiesService.getRelativePath(rootPath, filePath) : '';
            }
            if (filterPath === data[0].name) filterPath = '';
            let hasChild: boolean;
            if (stats.isDirectory()) {
                const dirStats: string[] = await fsPromises.readdir(filePath);
                hasChild = dirStats.some(async (name) => {
                    try {
                        const dir = await fsPromises.lstat(filePath + '/' + name);
                        return dir.isDirectory();
                    } catch {
                        return false;
                    }
                });
            } else {
                hasChild = false;
            }
            // Potentially check permissions here as well
            return {
                name: path.basename(filePath),
                size: this.utilitiesService.getSize(stats.size),
                isFile: stats.isFile(),
                dateModified: stats.ctime,
                dateCreated: stats.mtime,
                type: path.extname(filePath),
                filterPath,
                hasChild,
            };
        } catch (err) {
            Log.error(`Failed to get content: ${err}`);
        }
    }

    /**
     * Gget information about a file from the file system
     * @param file which file to look at
     * @param filePath full path to file
     * @param rootPath rootPath with projects or laboratories
     */
    private async _readFileInformation(file: string, filePath: string, rootPath: string): Promise<FileInformation> {
        try {
            const fileStats = await fsPromises.stat(file);
            let hasChild = false;
            let size = 0;
            if (fileStats.isDirectory()) {
                const dirStats: string[] = await fsPromises.readdir(file);
                await Promise.all(
                    dirStats.map(async (name) => {
                        try {
                            const dir = await fsPromises.stat(file + '/' + name);
                            size += dir.size;
                            if (dir.isDirectory()) hasChild = true;
                        } catch (err) {
                            Log.error(err);
                        }
                    }),
                );
            } else {
                size = fileStats.size;
            }
            // Check Permissions if we're using them here
            return {
                name: path.basename(filePath + file),
                size: this.utilitiesService.getSize(size),
                isFile: fileStats.isFile(),
                dateModified: fileStats.mtime,
                dateCreated: fileStats.ctime,
                filterPath: this.utilitiesService.getRelativePath(rootPath, filePath),
                type: path.extname(file),
                hasChild,
            };
        } catch (err) {
            Log.error(`Failed to read directory: ${err}`);
            throw err;
        }
    }

    /**
     * Gets information about files from file system
     * @param tokenFilePath path to file from token, contains /projectId/sampleId/
     * @param filePath path to the file or (root, sd_ui_sample, sd_ui_local)
     * @param data Information about the file from file manager
     * @param target where to look for files project or labs
     */
    public async readFiles(tokenFilePath: string, filePath: string, request: any): Promise<ReadResponse> {
        Log.info('check', tokenFilePath, filePath);
        const { data, target, sampleName } = request;
        try {
            switch (filePath) {
                case 'root':
                    return this._setupRoot(sampleName);
                case 'sd_ui_sample':
                    return await this._readFiles(
                        target,
                        path.join(this._dataManagerService.fsRoot[target], tokenFilePath),
                        data.map((i) => ({ ...i, filterPath: i.sample_id })),
                    );
                case 'sd_ui_local':
                    return await this._readFiles(target, this._dataManagerService.fsRoot[target], [{ filterPath: '/' } as any]);
                default:
                    let relativePath: string;
                    if (target === 'local') {
                        relativePath = path.join(this._dataManagerService.fsRoot[target], filePath);
                    } else {
                        relativePath = path.join(this._dataManagerService.fsRoot[target], tokenFilePath);
                    }
                    return await this._readFiles(target, relativePath, data);
            }
        } catch (err) {
            Log.error(`Failed to read files: ${err}`);
            const error: FileManagerError = {
                message: err,
                code: '401',
                fileExists: [],
            };
            return {
                cwd: null,
                files: [],
                error,
            };
        }
    }

    /**
     * Creates the Initial Sample and Files setup for the UI
     * @param sampleName Name of the Current Sample Alias
     * @returns A Read request { cwd, files }
     */
    private _setupRoot(sampleName: string): { cwd: any; files: FileInformation[] } {
        const cwd = { name: 'Files' };
        const files: FileInformation[] = [];
        const sample: FileInformation = {
            name: sampleName || 'Current Sample',
            dateCreated: undefined,
            dateModified: undefined,
            filterPath: 'sd_ui_sample',
            hasChild: false,
            isFile: false,
            size: '',
            type: 'folder',
        };
        files.push(sample);
        if (!!this._dataManagerService.localFiles && this._dataManagerService.localFiles.show) {
            const local: FileInformation = {
                name: this._dataManagerService.localFiles.name || 'Remote Files',
                dateCreated: undefined,
                dateModified: undefined,
                filterPath: 'sd_ui_local',
                hasChild: true,
                isFile: false,
                size: '',
                type: 'folder',
            };
            files.push(local);
        }

        return { cwd, files };
    }

    /**
     * Read the files on the system, either at Local location or in the Sample Directory
     * @param target Where are we manipulating files
     * @param filePath file location
     * @param data Stored CWD
     * @returns A Read request { cwd, files }
     */
    private async _readFiles(target: string, filePath: string, data: FileInformation[]): Promise<{ cwd: any; files: FileInformation[] }> {
        const filesList = await this._getFiles(filePath);
        const files: FileInformation[] = [];
        const cwdFiles = await this._fileManagerDirectoryContent(data, filePath, this._dataManagerService.fsRoot[target]);
        cwdFiles.target = target;
        cwdFiles.name = path.basename(filePath);
        for (const file of filesList) {
            const currentPath = path.join(filePath, file);
            let fileInfo: FileInformation;
            try {
                fileInfo = await this._readFileInformation(currentPath, filePath, this._dataManagerService.fsRoot[target]);
            } catch {
                fileInfo = {
                    name: path.basename(filePath + currentPath),
                    dateCreated: new Date(),
                    dateModified: new Date(),
                    filterPath: this.utilitiesService.getRelativePath(this._dataManagerService.fsRoot[target], filePath),
                    hasChild: false,
                    isFile: false,
                    size: 0,
                    type: '',
                    sample_id: '',
                };
            }
            fileInfo.target = target;
            files.push(fileInfo);
        }
        return {
            cwd: cwdFiles,
            files,
        };
    }

    /**
     * Returns file details about the selected files
     * @param names list of files to extract data about
     * @param data data to parse about the file
     */
    public getFileDetails(names: string[], data: FileInformation[]) {
        const response: DetailsResponse = {};
        const fileDetails: FileDetails = {
            name: data[0].name,
            size: data[0].size as string,
            isFile: data[0].isFile,
            modified: data[0].dateModified,
            created: data[0].dateCreated,
            type: data[0].type,
            location: data[0].filterPath,
        };
        if (data.length === 1) {
            response.details = fileDetails;
        } else {
            let size = 0;
            const nameList = names
                .map((name) => {
                    if (name.split('/').length > 0) {
                        return name.split('/')[name.split('/').length - 1];
                    } else {
                        return name;
                    }
                })
                .join(', ');
            data.forEach((item) => (size += this.utilitiesService.reverseSize(item.size)));
            response.details = {
                ...fileDetails,
                size: this.utilitiesService.getSize(size),
                name: nameList,
                multipleFiles: true,
            };
        }
        return response;
    }

    /**
     * Upload a file to the satellite from your computer
     * @param fileData information about the file to upload
     * @param name where to put file
     * @param target where to put file project or labs
     */
    public async uploadToSatellite(fileData: UploadInformation, name: string): Promise<string> {
        try {
            Log.info(`Attempting to upload ${fileData.originalname}`);
            const newLocation = path.join(this._dataManagerService.fsRoot['projects'], name, fileData.originalname);
            // const exists = await this.utilitiesService.directoryExists(location);
            // if (!exists) await fsPromises.mkdir(location);
            await fsPromises.rename(fileData.path, newLocation);
            Log.debug(`Successfully uploaded ${fileData.originalname}`);
            return 'Success';
        } catch (err) {
            Log.error(`Error uploading file: ${err}`);
            throw err;
        }
    }

    /**
     * Delete a file from satellite
     * @param tokenFilePath path to file from token, contains /projectId/sampleId/
     * @param name name of file to delete
     * @param target where to look for files project or labs
     */
    public async deleteFromSatellite(tokenFilePath: string, name: string, target: string): Promise<any> {
        try {
            await fsPromises.unlink(path.join(this._dataManagerService.fsRoot[target], tokenFilePath, name));
            return;
        } catch (err) {
            throw err;
        }
    }

    /**
     * Rename a file, could also be used to move if we wanted
     * @param tokenFilePath path to file from token, contains /projectId/sampleId/
     * @param oldLocation where file was
     * @param newLocation where to put file
     * @param target where to look for files project or labs
     */
    public async renameOnSatellite(tokenFilePath: string, oldLocation: string, newLocation: string, target: string): Promise<any> {
        try {
            const newPath = path.join(this._dataManagerService.fsRoot[target], tokenFilePath, newLocation);
            const oldPath = path.join(this._dataManagerService.fsRoot[target], tokenFilePath, oldLocation);
            await fsPromises.rename(oldPath, newPath);
            return;
        } catch (err) {
            throw err;
        }
    }

    /**
     * Moves files from local path location to project sample directory
     * @param currentLoc Current Location of files in Local Path directory
     * @param tokenPath Project ID and Sample ID joined
     */
    public async copyLocalFiles(currentLoc: string, tokenPath: string) {
        const split = currentLoc.split('/');
        const fileName = split[split.length - 1];
        const finalLocation = path.join(this._dataManagerService.fsRoot['projects'], tokenPath, fileName);
        try {
            await fsPromises.mkdir(path.join(this._dataManagerService.fsRoot['projects'], tokenPath), { recursive: true });
        } catch (err) {
            Log.debug('Directory Already Exists: ', err.message);
        }
        try {
            if (!currentLoc.includes(this._dataManagerService.fsRoot['local'])) throw new Error('Moving from outside local root');
            Log.info(`Copying from ${currentLoc} to ${finalLocation}`);
            await fsPromises.copyFile(currentLoc, finalLocation);
        } catch (err) {
            Log.error(`Error moving ${currentLoc} to ${finalLocation}`);
            Log.error(err.message);
        }
    }
