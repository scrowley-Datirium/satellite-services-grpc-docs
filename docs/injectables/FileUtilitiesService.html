import { Injectable } from '@nestjs/common';
import fs = require('fs');

const fsPromises = fs.promises;

@Injectable()
export class FileUtilitiesService {
    /**
     * Gets just the relative path from the file manager entry point
     * @param rootDirectory the root directory
     * @param fullPath the full path to the file
     */
    public getRelativePath(rootDirectory: string, fullPath: string): string {
        if (rootDirectory.substring(rootDirectory.length - 1) == '/') {
            if (fullPath.indexOf(rootDirectory) >= 0) {
                const relativePath = fullPath.substring(rootDirectory.length - 1).replace('//', '/');
                return relativePath[relativePath.length - 1] !== '/' ? relativePath + '/' : relativePath;
            }
        } else if (fullPath.indexOf(rootDirectory + '/') >= 0) {
            return fullPath.substring(rootDirectory.length + 1);
        } else {
            return '';
        }
    }

    /**
     * Get the size of the file in a string
     * @param size size of file in bytes
     */
    public getSize(size: number): string {
        let hz;
        if (size < 1024) hz = size + ' B';
        else if (size < 1024 * 1024) hz = (size / 1024).toFixed(2) + ' KB';
        else if (size < 1024 * 1024 * 1024) hz = (size / 1024 / 1024).toFixed(2) + ' MB';
        else hz = (size / 1024 / 1024 / 1024).toFixed(2) + ' GB';
        return hz;
    }

    /**
     * Make sure size is a number so that it combines easily
     * @param size string or potentially a number
     */
    public reverseSize(size: string | number): number {
        if (typeof size === 'number') return size;
        const num = parseInt(size.split(' ').shift());
        if (size.includes('KB')) return num * 1024;
        if (size.includes('MB')) return num * 1024 * 1024;
        if (size.includes('GB')) return num * 1024 * 1024 * 1024;
        return num;
    }


    /**
     * Does a directory already exist?
     * @param location location of directory
     */
    public async directoryExists(location: string): Promise<boolean> {
        try {
            await fsPromises.stat(location);
            return true;
        } catch {
            return false;
        }
    }
}
