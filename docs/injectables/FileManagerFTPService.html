import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { AccessOptions, Client as FTPClient, FileInfo } from 'basic-ftp';
import aria2 = require('aria2');
import path = require('path');

import Log from '../lib/logger';
import { Aria2cResponse, FileInformation, FTPInformation, ReadResponse } from 'src/interfaces';
import { FileUtilitiesService } from './fileUtilites.service';

import { createTunnel, closeTunnel } from 'proxy-chain';
import { connectForPassiveTransfer, parsePasvResponse } from 'basic-ftp/dist/transfer';

@Injectable()
export class FileManagerFTPService {
    private _ftpClient = new FTPClient();
    private _ftpOptions: AccessOptions;
    private _aria: any;

    constructor(private utilitiesService: FileUtilitiesService, private configService: ConfigService) {
        this._aria = new aria2(this.configService.get('download.aria2'));
    }
    /**
     * Gets recursive size of directory from ftp
     * @deprecated Takes a lot of time to run maybe use maybe not
     * @param filePath Path to file
     */
    private async _getDirectorySizeFTP(filePath) {
        try {
            let size = 0;
            const fileList: FileInfo[] = await this._ftpClient.list(filePath);
            for (const file of fileList) {
                if (!file.isDirectory) {
                    size += file.size;
                } else {
                    const location = filePath === '/' ? filePath + file.name : `${filePath}/${file.name}`;
                    size += await this._getDirectorySizeFTP(location);
                }
            }
            return size;
        } catch (err) {
            Log.error(`Error getting directory size: ${err}`);
            throw err;
        }
    }

    /**
     * Gets information about ftp files
     * @param filePath location of the file
     * @param ftpInformation ftp login information
     */
    public async readFTPFiles(filePath: string, ftpInformation: FTPInformation): Promise<ReadResponse> {
        let tunneledCommands = null;
        let tunneledData = null;
        try {
            const ftpOptions: AccessOptions = {
                host: ftpInformation.address,
                user: ftpInformation.username,
                password: ftpInformation.password,
            };
            this._ftpOptions = ftpOptions;
            if (process.env.ftp_proxy) {
                tunneledCommands = await createTunnel(
                    process.env.ftp_proxy,
                    `${ftpInformation.address}:21`,                          // UI doesn't have option to set port, so we use default 21
                    { hostname: 'localhost' }
                );
                Log.debug('Create tunnel for FTP commands', tunneledCommands);
                this._ftpClient.prepareTransfer = async function (ftp) {
                    const res = await ftp.request("PASV");
                    const target = parsePasvResponse(res.message);
                    tunneledData = await createTunnel(
                        process.env.ftp_proxy,
                        `${ftpInformation.address}:${target.port}`,
                        { hostname: 'localhost' }
                    );
                    Log.debug('Create tunnel for FTP data', tunneledData);
                    await connectForPassiveTransfer(
                        tunneledData.split(':')[0],
                        tunneledData.split(':')[1],
                        ftp
                    )
                    return res
                }
                const tunneledFtpOptions: AccessOptions = {
                    host: tunneledCommands.split(':')[0],
                    port: tunneledCommands.split(':')[1],
                    user: ftpInformation.username,
                    password: ftpInformation.password,
                };
                await this._ftpClient.access(tunneledFtpOptions);
            } else {
                await this._ftpClient.access(this._ftpOptions);
            }
            const cwd: FileInformation = {
                filterPath: filePath,
                name: filePath,
                hasChild: true,
                dateCreated: new Date(),
                dateModified: new Date(),
                isFile: false,
                size: '0',
                type: '',
            };
            const fileList: FileInfo[] = await this._ftpClient.list(filePath);
            const files: FileInformation[] = [];
            for (const file of fileList) {
                let size: number;
                if (!file.isDirectory) size = file.size;
                else {
                    size = file.size;
                    // TODO: Do we want to use this?
                    // This slows the ftp connection down by a few seconds because
                    // it recursively gets the file size of everything in the folder
                    // without this it just assigns a file size of 4096 to folders
                    // const location = filePath === '/' ? filePath + file.name : `${filePath}/${file.name}`;
                    // size = await this._getDirectorySizeFTP(location);
                }
                files.push({
                    name: file.name,
                    filterPath: filePath === '/' ? filePath : `${filePath}/`,
                    hasChild: file.isDirectory,
                    dateModified: new Date(file.rawModifiedAt),
                    dateCreated: new Date(file.rawModifiedAt),
                    isFile: file.isFile,
                    size: this.utilitiesService.getSize(size),
                    type: '',
                });
            }
            return {
                cwd,
                files,
            };
        } catch (err) {
            Log.error(`Failed to connect to ftp: ${err}`);
        } finally {
            this._ftpClient.close();
            if (tunneledCommands !== null){
                await closeTunnel(tunneledCommands, true);
                Log.debug('Close tunnel for FTP commands', tunneledCommands);
            }
            if (tunneledData !== null){
                await closeTunnel(tunneledData, true);
                Log.debug('Close tunnel for FTP data', tunneledData);
            }
        }
    }

    /**
     * Download a file from FTP to the satellite
     * @param data information from the file manager about the file
     * @param path path to the file
     * @param sampleId Sample Id
     */
    public async downloadFileFromFTP(data: FileInformation[], filePath: string, sampleId: string): Promise<Aria2cResponse[]> {
        try {
            Log.debug('Attempting to download to satellite');
            await this._aria.open();
            Log.debug('Aria2c connection open');
            const results: Aria2cResponse[] = [];
            for (const item of data) {
                const tempPath = path.resolve('/', filePath, sampleId);
                const directory = path.join(this.configService.get('systemRoot'), 'projects', tempPath);
                const location = item.filterPath + item.name;
                const guid = await this._downloadToSatellite(directory, item.name, location);
                results.push({ guid, location: filePath + '/' + item.name });
            }
            Log.debug('Downloading Started');
            // Stop tracking the downloads once they are all complete
            return results;
        } catch (err) {
            Log.error(`Failed to download: ${err}`);
            throw err;
        }
    }

    /**
     * Start the aria2c process of downloading the file
     * @param directory temporary location to download onto the fs
     * @param filename name of the file
     * @param location where it is on the ftp server
     */
    private async _downloadToSatellite(directory: string, filename: string, location: string): Promise<string> {
        const ftpLocation = `ftp://${this._ftpOptions.user}:${this._ftpOptions.password}@${this._ftpOptions.host}${location}`;
        try {
            const guid = await this._aria.call('addUri', [ftpLocation], {
                dir: directory,
                out: filename,
            });
            return guid;
        } catch (err) {
            Log.error(`Error with aria2c: ${err}`);
        }
    }
}
