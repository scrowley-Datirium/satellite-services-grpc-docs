import { Module } from '@nestjs/common';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { MulterModule } from '@nestjs/platform-express';
import { AuthModule } from 'src/auth/auth.module';
import path = require('path');

import { FileManagerAPIController } from './fileManagerAPI.controller';
import { FileManagerAPIService } from './fileManagerAPI.service';
import { FileManagerFTPService } from './fileManagerFTP.service';
import { FileUtilitiesService } from './fileUtilites.service';
import { DataManagerModule } from 'src/dataManager/dataManager.module';

@Module({
    imports: [
        AuthModule,
        MulterModule.registerAsync({
            imports: [ConfigModule],
            useFactory: async (configService: ConfigService) => ({
                dest: configService.get('tmp'),
            }),
            inject: [ConfigService],
        }),
        DataManagerModule,
    ],
    providers: [FileManagerAPIService, FileUtilitiesService, FileManagerFTPService],
    controllers: [FileManagerAPIController],
    exports: [FileManagerAPIService],
})
export class FileManagerAPIModule {}
