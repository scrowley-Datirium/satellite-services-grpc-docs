import { forwardRef, Inject, Injectable } from '@nestjs/common';
import { createMachine, interpret, spawn } from 'xstate';
import { Subject } from 'rxjs';
import pm2Types from 'pm2';
import network = require('network');
// eslint-disable-next-line @typescript-eslint/no-var-requires
const pm2: typeof pm2Types = require('pm2');

import { stateLogger } from '../lib/logger';
import defaultMachineConfig from '../lib/stateMachine';
import { OptionTypes, ProcessedTree, SatelliteMessage, ServiceContext, ServiceEvent, ServiceInterpreter, ServiceMachine, ServiceState } from '../interfaces';
import { AppService } from '../app.service';

@Injectable()
export class StateMachinesService {
    // Subject that sends Event to other State Machines
    //  type is the Event
    //  to is the name of the State Machine
    public sendEvent$ = new Subject<{ to: string; type: string }>();

    // Subject that sends a message to API-Sync
    private apisync$ = new Subject<SatelliteMessage>();
    // Allows custom scripts to access pm2
    private _pm2 = pm2;
    // Needed to get ips
    private _network = network;

    constructor(
        @Inject(forwardRef(() => AppService))
        private _appService: AppService,
    ) {
        this.apisync$.subscribe((msg) => {
            this._appService.sendToAPISync(msg);
        });
    }

    /**
     * Assemble the Finite State Machine for a Service
     * @param tree A tree node from API-Sync
     * @param script The updated Script Location
     * @param child False if Root Node
     * @param children Array of Child Machines to insert
     * @returns A Finite State Machine representing a Service
     */
    public createStateMachine(tree: ProcessedTree, script: string, child: boolean, children: ServiceMachine[]): ServiceMachine {
        const { name, options, errorresponse, outresponse, apimachine } = tree;
        // Create a custom logger for each state machine so that they are easier to distinguish
        const logger = stateLogger(name);
        const updatedDefaultConfig = { ...defaultMachineConfig };
        // If any global on Events have been declared add them to the state machine
        if (tree.stateMachine.on) {
            updatedDefaultConfig.on = { ...updatedDefaultConfig.on, ...tree.stateMachine.on };
        }

        let busresponse = (args: any) => null;
        if (tree.busresponse) {
            busresponse = (args: any) => {
                try {
                    // Call the function body with all of the helper arguments and bind it to this context
                    new Function('args', 'children', 'logger', tree.busresponse).call(this, args, children, logger);
                } catch (err) {
                    logger.error(`Failed to run action ${tree.busresponse}: ${err}`);
                }
            };
        }

        const updatedActions = {};
        // If any actions have been declared parse them correctly
        if (tree.stateMachine.actions) {
            const actions = tree.stateMachine.actions;
            Object.keys(actions).forEach(
                (k) =>
                    (updatedActions[k] = (ctx: ServiceContext) => {
                        try {
                            // Call the function body with all of the helper arguments and bind it to this context
                            new Function('ctx', 'children', 'logger', actions[k]).call(this, ctx, children, logger);
                        } catch (err) {
                            logger.error(`Failed to run action ${k}: ${err}`);
                        }
                    }),
            );
        }

        const updatedGuards = {};
        // If any guards have been declared parse them correctly
        if (tree.stateMachine.guards) {
            const guards = tree.stateMachine.guards;
            Object.keys(guards).forEach(
                (k) =>
                    (updatedGuards[k] = (ctx: ServiceContext) => {
                        try {
                            // Call the function body with all of the helper arguments and bind it to this context
                            return new Function('ctx', 'children', 'logger', guards[k]).call(this, ctx, children, logger);
                        } catch (err) {
                            logger.error(`Failed to run guard ${k}: ${err}`);
                        }
                    }),
            );
        }
        // If any states have been declared add them to the state machine
        if (tree.stateMachine.states) {
            const updatedStates = updatedDefaultConfig.states;
            Object.keys(tree.stateMachine.states).forEach((key) => {
                const state = tree.stateMachine.states[key];
                if (state.overwrite) {
                    updatedStates[key] = state;
                    return;
                }
                const finalState = updatedDefaultConfig.states[key] || {};
                const finalEntry = finalState.entry || [];
                const stateEntry = state.entry || [];
                const finalExit = finalState.exit || [];
                const stateExit = state.exit || [];
                finalState.entry = [...new Set([...finalEntry, ...stateEntry])];
                finalState.exit = [...new Set([...finalExit, ...stateExit])];
                const finalOn = finalState.on || {};
                if (state.on) {
                    Object.keys(state.on).forEach((k) => {
                        const onState = state.on[k];
                        const onFinal = finalOn[k] || { actions: [] };
                        const onFinalActions = onFinal.actions || [];
                        const onStateActions = onState.actions || [];
                        onFinal.actions = [...new Set([...onFinalActions, ...onStateActions])];
                        if (onState.target) onFinal.target = onState.target;
                        if (onState.cond) onFinal.cond = onState.cond;
                        finalOn[k] = onFinal;
                    });
                }
                finalState.on = finalOn;
                updatedStates[key] = finalState;
            });
            updatedDefaultConfig.states = updatedStates;
        }
        const machine = createMachine<ServiceContext, ServiceEvent, ServiceState>(
            {
                // Start with the default config
                ...updatedDefaultConfig,
                id: name,
                context: () => {
                    const childrenSpawn: any = {};
                    // Spawn all of the children within the machine itself, so that they have the correct relationship
                    children.forEach((i) => {
                        childrenSpawn[i.machine.id] = spawn(i, i.machine.id);
                    });
                    return {
                        name,
                        script,
                        namespace: 'scidap-satellite',
                        notChild: !child,
                        options: { ...options, script },
                        onlyMachine: tree.onlyMachine,
                        errorresponse,
                        outresponse,
                        stdin: null,
                        apimachine,
                        busresponse,
                        ...childrenSpawn,
                    };
                },
            },
            {
                guards: {
                    // Guard the Offline to Started route so that a child must pass through Ready
                    notChild: (ctx: ServiceContext) => ctx.notChild,
                    ...updatedGuards,
                },
                actions: {
                    // Send the STOPPED event to any child when the parent stops
                    sendStopToChild: (ctx) => {
                        if (!children.length) return;
                        logger.info('Sending Stop to children: ', children.map((i) => i.machine.id).join(', '));
                        children.forEach((child) => {
                            ctx[child.machine.id].send('STOPPED');
                        });
                    },
                    // Send the PARENT_ONLINE event to any child when the parent is online
                    sendReadyToChild: (ctx) => {
                        if (!children.length) return;
                        logger.info('Sending Ready to children: ', children.map((i) => i.machine.id).join(', '));
                        children.forEach((child) => {
                            ctx[child.machine.id].send('PARENT_ONLINE');
                        });
                    },
                    ...updatedActions,
                },
            },
        );
        return machine;
    }

    /**
     * Pass a FSM through the interpreter so that we can use it
     * @param machine An assembled Finite State Machine
     * @returns An Interpreted Machine
     */
    public interpretStateMachine(machine: ServiceMachine): ServiceInterpreter {
        const interpreted = interpret<ServiceContext, ServiceState, ServiceEvent>(machine);
        return interpreted;
    }
}
