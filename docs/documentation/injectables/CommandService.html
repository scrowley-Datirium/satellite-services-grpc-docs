import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { join } from 'path';
import { promises as fsPromises } from 'fs';

import { SatelliteTask, ServiceCommand, ServiceInterpreter } from './interfaces';
import Log from './lib/logger';
import { randomName } from './lib/utilities';
import { Pm2Service } from './pm2/pm2.service';

@Injectable()
export class CommandService {
    constructor(private _pm2Service: Pm2Service) {}
    /**
     * Send a command to a Machine
     * @param id Process Id
     * @param machine Machine to send task to
     * @param command Command to be executed
     * @returns A Promise after the command has been sent
     */
    public async forMachine(id: number, machine: ServiceInterpreter, command: ServiceCommand): Promise<boolean> {
        if (command.bus) {
            try {
                await this._pm2Service.sendPm2Bus(id, command.bus);
            } catch (err) {
                Log.error(`Failed to send command to ${command.target} through bus: ${err}`);
            }
        }
        if (command.stdin) {
            for (const line of command.stdin) {
                try {
                    await this._pm2Service.sendStdin(id, line);
                } catch (err) {
                    Log.error(`Failed to send command to ${command.target} through stdin: ${err}`);
                    Log.error(line);
                }
            }
        }
        // If there is a command to send to a machine
        if (command.command) {
            // Get a list of all of the possible on events for the target state machine
            //  This isn't strictly nessecary but without it, if you send an event that
            //  does not match one of these, it will just fail silently
            const onList = new Set<string>();
            const currConfig = machine.machine.config;
            // Find all the global on events for this machine
            Object.keys(currConfig.on).forEach((k) => onList.add(k));
            // Find all on events for every state
            //  We don't want to find just the active state on events because the state machine
            //  remembers events that don't apply and applies them when in the correct state
            Object.keys(currConfig.states).forEach((k) => {
                const s = currConfig.states[k];
                if (s.on) {
                    Object.keys(s.on).forEach((k) => onList.add(k));
                }
            });
            if (!onList.has(command.command)) {
                Log.error(`${command.target} does not have a command ${command.command}`);
                Log.debug('Command List:', [...onList].join(', '));
            }
            await machine.send({ type: command.command });
        }
        return true;
    }

    /**
     * Send a command to a Task
     * @param id Process Id
     * @param command Command to be executed
     * @returns A Promise after the command has been sent
     */
    public async forTask(id: string, command: ServiceCommand): Promise<boolean> {
        Log.info(`Sending command to ${command.target}`);
        if (command.bus) {
            try {
                await this._pm2Service.sendPm2Bus(id, command.bus);
            } catch (err) {
                Log.error(`Failed to send command to ${command.target} through bus: ${err}`);
            }
        }
        if (command.stdin) {
            try {
                await this._pm2Service.sendStdin(id, command.stdin);
            } catch (err) {
                Log.error(`Failed to send command to ${command.target} through stdin: ${err}`);
            }
        }
        return true;
    }
}
