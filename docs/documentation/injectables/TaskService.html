import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { join } from 'path';
import { promises as fsPromises } from 'fs';

import { SatelliteTask, ServiceInterpreter } from './interfaces';
import Log from './lib/logger';
import { randomName } from './lib/utilities';
import { Pm2Service } from './pm2/pm2.service';

@Injectable()
export class TaskService {
    // The folder where we write scripts and temporary logs
    //  In the settings.yml 'tmp' key
    private _tempFolder: string;

    constructor(private _configService: ConfigService, private _pm2Service: Pm2Service) {
        this._tempFolder = !!this._configService.get('tmp') ? this._configService.get('tmp') : '/tmp/scidap_satellite';
    }

    /**
     * Parse the Options and Set up Temporary Folders
     * @param task Task from API-Sync
     * @returns Locations and updated Options
     */
    public async processTaskForPm2(task: SatelliteTask): Promise<any> {
        const folderName = randomName(task.name);
        const errorLoc = join(this._tempFolder, folderName, `${task.name}-error.log`);
        const outLoc = join(this._tempFolder, folderName, `${task.name}-out.log`);
        const options = task.options || {};
        if (!options.name) options.name = task.name;
        options.output = outLoc;
        options.error = errorLoc;
        if (task.scriptstring) {
            const dirname = join(this._tempFolder, folderName);
            try {
                await fsPromises.mkdir(dirname);
            } catch (err) {
                Log.debug('Output Folder already exists');
            }
            try {
                const fileName = join(this._tempFolder, folderName, 'script');
                await fsPromises.writeFile(fileName, task.scriptstring);
                await fsPromises.chmod(fileName, 0o777);
                options.script = fileName;
            } catch (err) {
                const msg = `Error Creating Script ${task.name}: ${err}`;
                return { created: false, msg };
            }
        }
        const parsedOptions = this._pm2Service.parseOptions(options);
        return {
            created: true,
            options: parsedOptions,
            errorLoc,
            outLoc,
            folderName,
        };
    }
}
