import { forwardRef, HttpModule, Module } from '@nestjs/common';
import { ScheduleModule } from '@nestjs/schedule';
import { ClientProxyFactory, ClientsModule, Transport } from '@nestjs/microservices';
import { JwtModule } from '@nestjs/jwt';
import { credentials } from 'grpc';
import { join } from 'path';

import { AuthModule } from './auth/auth.module';
import { AppService } from './app.service';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { StateMachineModules } from './stateMachines/stateMachines.module';
import configuration from '../configuration';
import { TreeService } from './tree.service';
import { TaskService } from './task.service';
import { Pm2Module } from './pm2/pm2.module';
import { CommandService } from './command.service';

@Module({
    imports: [
        AuthModule,
        forwardRef(() => StateMachineModules),
        HttpModule,
        JwtModule.register({}),
        ScheduleModule.forRoot(),
        ConfigModule.forRoot({
            isGlobal: true,
            load: [configuration],
        }),
        forwardRef(() => Pm2Module),
    ],
    // This is the real world configuration that we will need to swap back to when we finish development
    providers: [
        {
            provide: 'API_SYNC_CONNECTION_SERVICE',
            useFactory: (configService: ConfigService) => {
                const sslConnection = configService.get('SSL_CONN') === 'true' ? true : false;
                return ClientProxyFactory.create({
                    transport: Transport.GRPC,
                    options: {
                        ...configService.get('gRPCOptions'),
                        credentials: sslConnection ? credentials.createSsl() : credentials.createInsecure(),
                        maxSendMessageLength: 15 * 1024 * 1024,
                        maxReceiveMessageLength: 15 * 1024 * 1024,
                    },
                });
            },
            inject: [ConfigService],
        },
        ConfigService,
        AppService,
        TreeService,
        TaskService,
        CommandService,
    ],
    exports: [AppService],
})
export class AppModule {}
