import { UrlWithStringQuery } from 'url';

export interface ProcessedData {
    cwlFile?: string;
    sample?: any;
    taskId: string;
    dagId?: string;
    downloads?: any[];
}

export interface BaseModuleInterface {
    getInfo(): {
        moduleId: string;
        caption: string;
        type: string;
    };

    getFile(fileUrl: UrlWithStringQuery, userId?: string, sampleId?: string): FileData;
}

export interface Info {
    moduleId: string;
    caption?: string;
    type?: string;
    mainUrl?: string;
    loginUrl?: string;
    viewListUrl?: string;
    downloadUrl?: string;
    private_key?: string;
    private_iv?: string;
    encryptKey?: string;
    token?: string;
    protocol?: string;
    collection?: any;
    publication?: string;
    refreshSessionInterval?: number;
}

export interface FileData {
    url: string;
    basename: string;
    header: string;
}

export interface DownloadObject {
    uri: string;
    path: string;
    header: string;
    sampleId: string;
    projectId: string;
    userId: string;
    fileId: string;
    inputKey?: string;
    token: string;
    downloaded: boolean;
    taskId: string;
    fileName: string;
    gid?: string;
}

export interface AriaDownloadObject extends DownloadObject {
    gid: string;
}

export interface DownloadResults {
    status: string;
    fileId?: string;
    taskId?: string;
    progress?: string;
    error?: boolean;
    path?: string;
    size?: number;
    message?: string;
    sampleId?: string;
}

export interface Aria2cProgressReport {
    status: string;
    completedLength: number;
    totalLength: number;
    files: AriaFileInfo[];
}

export interface MergedDownloadObject {
    aria2: Aria2cProgressReport;
    data: AriaDownloadObject;
}

export interface AriaFileInfo {
    path: string;
    uris: {
        uri: string;
    }[];
}
export interface AirflowReportToAPI {
    state: string;
    dagId: string;
    runId: string;
    progress: number;
    error: string;
}

export interface AirflowSuccessToAPI {
    dagId: string;
    runId: string;
    progress: number;
    results: any;
}

export interface SatelliteJWT {
    userId: string;
    iat: number; //in seconds
    exp: number; //in seconds
    aud: string; //satellite id
    sub: string; //satellite
}

export interface DownloadFailed {
    message: string;
    sampleId: string;
    taskId: string;
}
