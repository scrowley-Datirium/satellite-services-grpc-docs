import { WriteStream } from 'fs';
import { StartOptions } from 'pm2';
import { Interpreter, MachineConfig, StateMachine } from 'xstate';
import { OnOptions, OptionTypes, StateOptions } from '.';

export type ServiceInterpreter = Interpreter<ServiceContext, ServiceState, ServiceEvent>;
export type ServiceMachine = StateMachine<ServiceContext, ServiceState, ServiceEvent>;
export type ServiceMachineConfig = MachineConfig<ServiceContext, any, ServiceEvent>;

export type ServiceContext = BaseContext & ContextChildren;

interface BaseContext {
    name: string;
    script: string;
    notChild: boolean;
    options: StartOptions;
    namespace: 'scidap-satellite';
    replaceScript: boolean;
    onlyMachine: boolean;
    apimachine: string;
    outresponse: string;
    errorresponse: string;
    stdin: WriteStream;
    busresponse: (args: any) => void;
}

interface ContextChildren {
    [key: string]: Interpreter<ServiceContext, ServiceState, ServiceEvent>;
}

export type ServiceEvent =
    | { type: 'ERROR' }
    | { type: 'RESTART' }
    | { type: 'STOPPED' }
    | { type: 'STARTED' }
    | { type: 'PARENT_ONLINE' }
    | { type: string };

export type ServiceState =
    | {
          value: 'offline';
          context: ServiceContext;
      }
    | {
          value: 'ready';
          context: ServiceContext;
      }
    | {
          value: 'online';
          context: ServiceContext;
      }
    | {
          value: 'restarting';
          context: ServiceContext;
      }
    | {
          value: 'error';
          context: ServiceContext;
      };

export type ServiceStateValues = 'offline' | 'ready' | 'online' | 'restarting' | 'error';

export interface DefaultMachineConfig {
    id: string;
    initial: string;
    states: {
        [key: string]: StateOptions;
    };
    on: {
        [key: string]: OnOptions;
    };
}
