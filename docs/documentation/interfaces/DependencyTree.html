import { Metadata } from 'grpc';
import { Observable, Subject, Subscription } from 'rxjs';
import { StartOptions as PM2StartOptions } from 'pm2/types';

export interface APISyncService {
    streamQueue(upstream: Observable<SatelliteMessage>, metadata: Metadata): Observable<APISyncMessage>;
    commandStream(upstream: Observable<SatelliteOutput>): Observable<APISyncComplete>;
    fileStream(upstream: Observable<FileFromClient>): Observable<APISyncComplete>;
}

export interface APISyncMessage {
    taskId: string;
    satelliteid?: string;
    tasks: SatelliteTask[];
    trees: DependencyTree[];
    commands: ServiceCommand[];
    keepalive?: boolean;
}

export interface DependencyTree {
    children: DependencyTree[];
    options: CustomPm2StartOptions;
    statemachine?: UnProcessedStateMachineOptions;
    onlymachine?: boolean;
    apimachine?: string;
    outresponse?: string;
    errorresponse?: string;
    busresponse?: string;
}

export interface UnProcessedStateMachineOptions {
    states?: OptionTypes;
    on?: { [key: string]: OnOptions };
    guards?: { [key: string]: string };
    actions?: { [key: string]: string };
}

export interface ProcessedTree {
    name: string;
    children: ProcessedTree[];
    options: StartOptions;
    stateMachine: StateMachineOptions;
    onlyMachine: boolean;
    apimachine?: string;
    outresponse?: string;
    errorresponse?: string;
    busresponse?: string;
}

export interface StateMachineOptions {
    states?: OptionTypes;
    on?: { [key: string]: OnOptions };
    guards?: { [key: string]: string };
    actions?: { [key: string]: string };
}

export interface OptionTypes {
    [key: string]: StateOptions;
}

export interface StateOptions {
    on?: {
        [key: string]: OnOptions;
    };
    entry?: string[];
    exit?: string[];
    overwrite?: boolean;
}

export interface OnOptions {
    target?: string;
    actions?: string[];
    cond?: string;
}
export interface CustomPm2StartOptions extends StartOptions {
    // used for testing
    keepscript?: boolean;
    // Always use args as an array of strings instead of a string
    args?: string[];
    // Always use interpreter_args as an array of strings instead of a string
    interpreter_args?: string[];
    // Always use a number for max_memory_restart
    max_memory_restart?: number;

    // We can't set a variable field so if watch array is set we use it
    //  instead of watch
    watch?: boolean;
    watch_array?: string[];

    // We can't set a gRPC field to any so we need to stringify them on API sync
    //  and parse them on the satellite
    cron?: string;
    execute_command?: string;
    write?: string;
    source_map_support?: string;
    disable_source_map_support?: string;
}

export interface StartOptions extends PM2StartOptions {
    keepscript?: boolean;
    namespace?: string;
    autorestart?: boolean;
}

export interface SatelliteTask {
    name: string;
    apimachine: string;
    target: string;
    command?: string;
    options?: CustomPm2StartOptions;
    streamresponse?: boolean;
    scriptstring?: string;
    outresponse?: string;
    errorresponse?: string;
    savelogs?: boolean;
}

/* Satellite's Response, Including which part of the Satellite the message
 * originated from, type will be how we process it */
export interface SatelliteMessage {
    target?: string;
    taskid?: string;
    sender?: string;
    type?: string;
    messages?: { action: string; body: string }[];
    keepalive?: boolean;
}

}
