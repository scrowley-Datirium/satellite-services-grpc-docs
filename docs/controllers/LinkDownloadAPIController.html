import { Controller, Get, Param, Req, Res, UseGuards } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { Response } from 'express';
import mime = require('mime-types');
import path = require('path');

import Log from '../lib/logger';
import { FileManagerAPIService } from '../fileManagerAPI/fileManagerAPI.service';
import { GetAuthGuard } from './getAuth.guard';
import { LinkDownloadAPIService } from './linkDownloadAPI.service';
import { DataManagerService } from 'src/dataManager/dataManager.service';

@Controller('download')
export class LinkDownloadAPIController {
    constructor(
        private _dataManagerService: DataManagerService,
        private jwtService: JwtService,
        private fileManagerAPIService: FileManagerAPIService,
        private linkDownloaderAPIService: LinkDownloadAPIService,
    ) {}

    /**
     * Request from visual plugins for file on satellite
     * @param req Request Object
     * @param res Response Object
     * @param token JWT Token should follow this format
     */
    @UseGuards(GetAuthGuard)
    @Get(':token/*')
    async partialDownload(@Req() req, @Res() res: Response, @Param('token') token: string) {
        let currentPath: string;
        let userID: string;
        try {
            let reqPath = path.resolve('/', req.url.replace(/\/download\/.+?\//, ''));
            reqPath = reqPath.split(this._dataManagerService.scidapHome).pop();
            currentPath = path.join('/projects/', reqPath);
            const decoded: any = this.jwtService.decode(token);
            // Remove any extra information from the end of the path
            const woExtra = path.extname(reqPath.split('?')[0]);
            // Lookup MimeType and if it isnt found check the given mimetypes from apisync
            const typeLookup = mime.lookup(woExtra);
            let contentType: string;
            if (!typeLookup) {
                contentType = this.linkDownloaderAPIService.mimeTypes[woExtra] || this.linkDownloaderAPIService.mimeTypes.default;
            } else {
                contentType = mime.contentType(typeLookup);
            }
            Log.info(`${decoded.aud} is attempting to download ${currentPath}`);
            // We have access to the auth service so we can use it to validate the token
            const result = await this.fileManagerAPIService.sendFile(currentPath, req);
            if (result.error) {
                throw result.error;
            }
            Log.debug(`Range returned ${result.headers['Content-Range']}`);
            Log.info('Successfully found file');
            // If the content-type isn't recognized use binary
            result.headers['Content-Type'] = contentType;
            if (path.extname(currentPath) === '.html') {
                result.headers['Content-Disposition'] = 'inline';
            }
            await result.send(res);
        } catch (err) {
            Log.error(`${userID} failed to download ${currentPath}`);
            res.status(500);
            res.send(err);
        }
    }
}
