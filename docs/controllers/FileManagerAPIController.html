import { Body, Controller, HttpException, HttpStatus, Post, UploadedFile, UseGuards, UseInterceptors, Get } from '@nestjs/common';
import { FileInterceptor } from '@nestjs/platform-express';
import { diskStorage } from 'multer';
import path = require('path');
import fs = require('fs');

import Log from '../lib/logger';
import {
    Aria2cResponse,
    UploadInformation,
    DownloadToSatelliteRequest,
    UploadRequest,
    FileInformation,
    ReadRequest,
    FileManagerToken,
    SatelliteJWT,
} from '../interfaces';
import { FileManagerAPIService } from './fileManagerAPI.service';
import { FileManagerFTPService } from './fileManagerFTP.service';
import { AuthService } from 'src/auth/auth.service';
import { AuthGuard } from 'src/auth/auth.guard';
import { MulterOptions } from '@nestjs/platform-express/multer/interfaces/multer-options.interface';
import { ConfigService } from '@nestjs/config';

@Controller('filemanager/api/filemanager')
export class FileManagerAPIController {
    constructor(
        private fileManagerAPIService: FileManagerAPIService,
        private fileManagerFTPService: FileManagerFTPService,
        private configService: ConfigService,
        private authService: AuthService,
    ) {}

    @UseGuards(AuthGuard) // Guards the route
    @Post('FileOperations')
    async fileOperations(@Body() readRequest: ReadRequest): Promise<any> {
        const { satelliteId, projectId, sampleId } = this.authService.decode(readRequest.token) as FileManagerToken;
        const tokenFilePath = path.resolve('/', projectId, sampleId + '');
        Log.info(`Request from file manager ${satelliteId}`);
        // Action for getDetails
        if (readRequest.action === 'details') {
            Log.debug(`Details for ${readRequest.names.join(' ')}`);
            const response = this.fileManagerAPIService.getFileDetails(readRequest.names, readRequest.data);
            return response;
        }
        // Not currently used
        // Action for copying files
        if (readRequest.action === 'copy') {
            Log.info('Copying local files');
            this.fileManagerAPIService.copyLocalFiles(readRequest.currentLocation, tokenFilePath);
            // CopyFiles(req, res, contentRootPath);
        }
        // Not currently used
        // Action for movinh files
        if (readRequest.action === 'move') {
            Log.debug('Move');
            // MoveFiles(req, res, contentRootPath);
        }
        // Not currently used
        // Action to create a new folder
        if (readRequest.action === 'create') {
            Log.debug('Create');
            // createFolder(req, res, contentRootPath + req.body.path, contentRootPath);
        }
        // Action to remove a file
        if (readRequest.action === 'delete') {
            Log.debug('Delete');
            try {
                if (readRequest.ftp) throw new Error('Cannot Delete on FTP');
                // if (readRequest.path !== '/uploads/') throw new Error('Can only delete uploads');
                // const name = path.join(readRequest.data[0].filterPath, readRequest.names[0]);
                const name = readRequest.names[0];
                await this.fileManagerAPIService.deleteFromSatellite(tokenFilePath, name, readRequest.target);
                const response = {
                    files: readRequest.data,
                    error: null,
                };
                return response;
            } catch (err) {
                Log.error(`Error with delete: ${err}`);
                const error = {
                    message: err.message,
                    code: '500',
                };
                const response = {
                    error,
                    files: readRequest.data,
                };
                return response;
            }
        }
        // Action to rename a file
        if (readRequest.action === 'rename') {
            Log.debug('Rename');
            try {
                if (readRequest.ftp) throw new Error('Cannot Rename on FTP');
                // if (readRequest.path !== '/uploads/') throw new Error('Can only rename uploads');
                const oldName = path.resolve('/', readRequest.data[0].filterPath, readRequest.name);
                const newName = path.resolve('/', readRequest.data[0].filterPath, readRequest.newName);
                await this.fileManagerAPIService.renameOnSatellite(tokenFilePath, oldName, newName, readRequest.target);
                const response = {
                    files: readRequest.data,
                    error: null,
                };
                return response;
            } catch (err) {
                Log.error(`Error with rename: ${err}`);
                const error = {
                    message: err.message,
                    code: '500',
                };
                const response = {
                    error,
                    files: readRequest.data,
                };
                return response;
            }
        }
        // Action to read a file
        if (readRequest.action === 'read') {
            Log.debug('Read');
            try {
                let filePath: string;
                if (readRequest.data.length) {
                    if (readRequest.data[0].filterPath.includes('sd_ui')) {
                        filePath = readRequest.data[0].filterPath;
                    } else {
                        filePath = path.join(readRequest.data[0].filterPath, readRequest.data[0].name);
                    }
                } else {
                    filePath = readRequest.path;
                }
                const response = readRequest.ftp
                    ? await this.fileManagerFTPService.readFTPFiles(filePath, readRequest.ftpInformation)
                    : await this.fileManagerAPIService.readFiles(tokenFilePath, filePath, readRequest);
                return response;
            } catch (err) {
                return err;
            }
        }
    }

    @UseGuards(AuthGuard) // Guards the route
    @Post('downloadtosatellite')
    async downloadToSatellite(@Body() downloadRequest: DownloadToSatelliteRequest): Promise<Aria2cResponse[]> {
        try {
            const { details, path } = downloadRequest;
            const { sampleId } = this.authService.decode(downloadRequest.token) as { sampleId: string };
            const response = await this.fileManagerFTPService.downloadFileFromFTP(details, path, sampleId);
            Log.debug(response);
            return response;
        } catch (err) {
            Log.error(`Error downloading to satellite: ${err}`);
            return err;
        }
    }

    // Multipart form data is process after guards so we have to manually guard the route

    @UseInterceptors(FileInterceptor('uploadFiles'))
    async uploadFile(@Body() uploadRequest: UploadRequest, @UploadedFile() file: UploadInformation) {
        Log.debug('Upload request');
        try {
            const jwt = uploadRequest.token;
            const decoded = this.authService.decode(jwt) as FileManagerToken;
            const tokenFilePath = path.resolve('/', decoded.projectId, decoded.sampleId + '');
            // const data: FileInformation = JSON.parse(uploadRequest.data);
            // if (uploadRequest.target === 'projects') {
            //     if (
            //         (data.filterPath === '' && decoded.projectId !== data.name) ||
            //         (data.filterPath !== '' && !data.filterPath.includes(decoded.projectId))
            //     ) {
            //         throw new Error(
            //             `Project Id not valid. Token: ${decoded.projectId} Data: ${data.filterPath === '' ? data.name : data.filterPath}`,
            //         );
            //     }
            // }
            if (decoded.satelliteId !== this.authService.satelliteId) {
                throw new Error('Satellite Id not valid');
            }

            // const currentDateTime = new Date();
            // if (new Date(decoded.iat * 1000) >= currentDateTime || new Date(decoded.exp * 1000) <= currentDateTime) {
            //     throw new Error('JWT not valid');
            // }

            if (process.env.NODE_ENV !== 'development') {
                await this.authService.verify(jwt);
            }
            const response = await this.fileManagerAPIService.uploadToSatellite(file, tokenFilePath);
            return response;
        } catch (err) {
            Log.error(`Error uploading: ${err}`);
            fs.unlink(file.path, (err) => {
                if (err) {
                    Log.error(`Failed to remove ${file.path}`);
                } else {
                    Log.debug(`Successfully removed ${file.path}`);
                }
            });
            throw new HttpException('Forbidden', HttpStatus.FORBIDDEN);
        }
    }
